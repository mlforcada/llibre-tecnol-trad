\begin{slide}{}
\textbf{\large Aplicacions de la TA /2}
\stepwise
{
\vstep{\concept{Preedició i postedició}: els professionals col·laboren amb el
  sistema de TA en aplicacions de disseminació:}
\begin{itemize}
\vstep{\item\concept{Preedició:} preparació del text per a evitar
  lèxic o construccions que donen problemes de traducció amb un
  sistema de traducció automàtica.}
\vstep{\item\concept{Postedició:} correcció del text traduït en brut
  per a fer-lo adequat al propòsit previst.}
\end{itemize}
\vstep{Conéixer bé \emph{com funciona} el sistema de TA ajuda molt en
 ambdues tasques.}
}

\stepwise
{
\vstep{Alternativa a la preedició: \emph{llenguatge controlat}.}
\begin{itemize}\setlength{\itemsep}{0pt}
\vstep{\item Els autors escriuen ja pensant en el tractament automatitzat

\vstep{\item S'eviten lèxic i construccions problemàtiques.}
\vstep{\item Es minimitza la postedició.}
\vstep{\item Consistència d'estil, comprensibilitat, mantenibilitat.}
\vstep{\item Però els autors l'han de conéixer i aplicar!}
\vstep{\item Se'ls pot ajudar amb eines informàtiques.}
\end{itemize}
}

\end{itemize}
}

\begin{slide}{}
\textbf{\large Formats de text /1}
\stepwise
{
\vstep{Un \emph{text informatitzat} és, com qualsevol porció de dades
informatitzada, una \emph{seqüència de bits}, és a dir, d'\emph{uns} i
\emph{zeros}: \texttt{000101010010100111101001010010\ldots}.}

\vstep{Els \emph{bits} van normalment en grups de 8 (\emph{bytes} o
\emph{octets}). Amb 8 bits es poden fer $2\times 2\times
2\times2\times2\times2\times2\times2\times2$}\vstep{$=2^8$}\vstep{$=256$~combinacions:}
\vstep{\texttt{00000000} (0),} \vstep{\texttt{00000001} (1),} \vstep{\texttt{00000010} (2),}
\vstep{\ldots, \texttt{11111111} (255).}

\vstep{Hi ha moltes maneres d'organitzar els octets per a emmagatzemar
textos. Molts problemes provenen de discrepàncies quant a la manera de
fer-ho.}
}

\end{slide}

%%%%%%%%%%%%%%%%%%%

\begin{slide}{}
\textbf{\large Formats de text /2}
\stepwise
{
\vstep{Dos aspectes importants: \concept{codificació} i
  \concept{format} propiament dit.}
\begin{itemize}\setlength{\itemsep}{0pt}
    \vstep{\item\concept{Codificació:} Assignació d'un \emph{codi} (una
    seqüència d'un o més \emph{octets} concreta) a cada caràcter
    possible de la llengua corresponent (per exemple: ``\texttt{a}''
    $\to$ \texttt{01100001} (97); ``\texttt{?}'' $\to$
    ``\texttt{00111111}'' (63), etc.)}
\end{itemize}
}
\end{slide}


%%%%%%%%%%%%%%%%%%%
\begin{slide}{}
\textbf{\large Formats de text /3}
\stepwise
{
\begin{itemize}\setlength{\itemsep}{0pt}
\vstep{\item\concept{Format propiament dit:} Els textos, a més de
      caràcters, contenen informació de format.}

\vstep{És necessària l'assignació de 
      \emph{codis} per a regular altres característiques del text:}
\setlength{\parsep}{0pt}
\begin{itemize}\setlength{\itemsep}{0pt}
  \vstep{\item Per a codificar l'aparença \emph{visual} o de
    presentació, per exemple, ``inici cursives'', ``final negretes'',
    ``lletra de 16 punts''), o} \vstep{\item Per a codificar
    l'\emph{estructura} (és a dir, l'organització del contingut, per
    exemple, ``títol de secció'', ``llista numerada'', ``nota a peu de
    pàgina'', ``fila d'una taula'', etc.).}
\end{itemize}
\end{itemize}
}
\end{slide}


%%%%%%%%%%%%%%%%%%%
%%%%%%%%%%%%%%%%%%%
\begin{slide}{}
\textbf{\large Formats de text /4}
\stepwise
{
\vstep{Codificacions d'1 octet (``unibyte''):}
\begin{itemize}\setlength{\itemsep}{0pt}
  \vstep{\item\concept{ASCII:} Assigna codis de 7 bits, del
    \texttt{0000000} (0) al \texttt{1111111} (127), (sobra un bit de
    l'octet) als caràcters anglesos (sense accents, etc.)}
  \vstep{\item\concept{ANSI o ISO:} \emph{familia} de codificacions que
    aprofiten els codis del 128 al 255 per a caràcters internacionals.
 
    En Europa occidental: \emph{ISO-8859-1} (o Latin-1); més recentment,
    \emph{ISO-8859-15} (o Latin-9; conté el símbol de l'euro)}

  \vstep{En Windows s'usa \emph{CP-1252} que és molt similar (però no
  idèntic) a l'ISO-8859-1.}
\end{itemize}
}
\end{slide}
%%%%%%%%%%%%%%%%%%%
\begin{slide}{}
  \textbf{\large Formats de text /5} \stepwise { \vstep{Codificacions
      de més d'1 octet (``multibyte'': japonés, xinés, coreà,
      devanagari\ldots :)}
\begin{itemize}\setlength{\itemsep}{0pt}
  \vstep{\item\concept{Unicode (ISO-10646):} Assigna codis de 31 bits
    (4 octets) i permet codificar $2^{31}=2~147~483~648$ caràcters.}
  \vstep{\item\concept{UTF-8:} Versió d'Unicode que només usa més d'un
    octet quan cal:} 
\begin{itemize}\setlength{\itemsep}{0pt}
\vstep{\item codis del 0 al 127: 1 octet (compatible amb  ASCII):}
\vstep{\item codis del 128 al 2047: 2 octets;}
\vstep{\item codis del 2048 al 65535: 3 octets, etc.}
\end{itemize}
\end{itemize}
}
\end{slide}

     \begin{center}

     \end{center}
 }
\end{itemize}

\textbf{\large Formats de text /8}

El problema \emph{wysiwyg} (``what you see is what you get'': ``el que veus és
el que obtindràs''):
\stepwise
{
\vstep{Les persones usen la \emph{presentació visual} per a comunicar
  \emph{l'estructura lògica} dels documents (a persones vidents!).}

\vstep{Els processadors de textos actuals ens mostren el document \emph{tal
    com quedarà imprés} mentres l'estem editant.}

\vstep{Sucumbim a la temptació de treballar directament sobre la
  \emph{presentació} (negretes, màrgens, tipus) en comptes de sobre
  l'\emph{estructura lògica} (seccions, títols, etc.)}

}

\begin{slide}{}
\textbf{\large Formats de text /10}

\stepwise
{
\vstep{Com? Usant \emph{estils}.}

\vstep{Marquem \emph{estructuralment} les parts (elements): \emph{títol de
    segon nivell}, \emph{text emfatitzat}, \emph{exemple}, etc.}

\vstep{I després assignem un \emph{estil} de presentació a cada part (per

  només cal canviar l'estil associat a l'element.}

}

      part més mecànica (mecanitzable) de la tasca de traducció.}
    
    \vstep{Però \concept{no podem esperar} ---per bo que siga--- que comprenga
      el text, resolga les ambigüitats sempre correctament i produïsca
      textos en una variant genuïna de la llengua meta}.
 }

\end{slide}

\begin{slide}{}
  \textbf{\large Per què és difícil la TA? /1} 

  \stepwise {
  \vstep{Els quatre problemes de la traducció automàtica (Arnold 2003):}
  \begin{enumerate}
   \vstep{\item El problema de l'anàlisi}
   \vstep{\item El problema de la síntesi}
   \vstep{\item El problema de la transferència}
   \vstep{\item El problema de la descripció}
  \end{enumerate}
  }

  
\end{slide}
\begin{slide}{}
  \textbf{\large Per què és difícil la TA? /2} 
  
  \textbf{El problema de l'anàlisi:} La forma no determina completament el
  contingut (la \emph{interpretació}). També s'anomena \emph{ambigüitat}:
  \begin{itemize}
  \item \emph{Portaven notícies de Grècia} (tema o procedència?)
  \item \emph{Ha venut les taronges que ha
  comprat a Joan} (Joan ven o compra?)
  \item \emph{Treballa en l'estudi que li han encarregat} (prepara un document
  o dissenya un taller d'artista?)
  \end{itemize}
\end{slide}


%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
\begin{slide}{}
  \textbf{\large Per què és difícil la TA? /3} 
  
  \textbf{El problema de la síntesi:} El contingut no determina completament
  la forma (hi ha més d'una manera de dir el mateix en qualsevol
  llengua):
  \begin{itemize}
  \item \emph{Quina hora és?}
  \item \emph{Com és de tard?} (\texttt{de}: \emph{Wie spät ist es?})
  \item \emph{Quines hores són} (\texttt{pt}: \emph{Que horas s\~{a}o?}
  \end{itemize}

  Els \emph{expedients} s'\emph{obrin} o s'\emph{inicien}?

  Les sessions es \emph{clouen}, es \emph{tanquen}, es \emph{rematen} o \emph{s'alcen}?

\end{slide}

%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
\begin{slide}{}
  \textbf{\large Per què és difícil la TA? /4} 
  
  \textbf{El problema de la transferència:} Les llengües divergeixen. És a
  dir, hi ha diferències irreductibles en la manera en que el mateix contingut
  s'expressa en llengües diferents:
  \begin{itemize}\itemsep 0ex
  \item \texttt{ca:} \emph{M'agrada nadar} (\emph{M'} objecte, \emph{agrada},
    verb, \emph{nadar} subjecte)
  \item \texttt{en:} \emph{I like swimming} (\emph{I} subjecte, \emph{like}
  verb, 
  \emph{swimming} objecte)
  \item \texttt{de:} \emph{Ich schwimme gern} (\emph{Ich} subjecte,
  \emph{schwimme}, verb,  \emph{gern}, adverbi)
  \end{itemize}
  Totes volen dir \texttt{produir\_plaer(agent=nadar(agent=jo),destinatari=jo)}
\end{slide}

%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%

%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
\begin{slide}{}
  \textbf{\large Per què és difícil la TA? /5} 
  
  \textbf{El problema de la descripció} (represa): construir un sistema de
  traducció automàtica comporta la gestió d'una gran quantitat de coneixement,
  que s'ha d'elicitar, aplegar, descriure, i representar en una forma útil i
  computable.
\end{slide}
